Coding strategy: Since each day has 288 intervals, NA-values are filled with the value “dailymean/288” 

for (i in 1:nrow(data)){
    if (is.na(data[i,1])) {
       data[i,1] = dailymean/288 # 288 is the number of #intervals per day
    }
}
